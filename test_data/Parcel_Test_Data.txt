--- TEST 1: List All Parcels ---
Method: GET
URL: /api/v1/parcels?page=1&limit=20
Headers:
  Authorization: Bearer <OPERATOR_TOKEN>

Assertions:
  - Status: 200
  - Body: { success: true, data: [...], meta: { page, limit, totalRows, totalPages } }
