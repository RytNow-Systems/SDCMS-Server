### Scan and Link AWB (OPERATOR)
Method: POST
URL: /api/v1/parcels/scan
Headers: { "Authorization": "Bearer {{operatorToken}}", "Content-Type": "application/json" }
Payload:
{
  "qrCode": "PDS-A1B2C3",
  "awbNumber": "AWB-OP-1001"
}
Assertions:
- res.status equals 200
- res.body.data.status equals "AWB Linked"
