--- TEST 5: Update Order (Blocked — Parcel ≥ AWB_LINKED) ---
Method: PUT
URL: /api/v1/orders/1
Headers:
  Authorization: Bearer <OPERATOR_TOKEN>
  Content-Type: application/json

Pre-condition: At least one parcel in this order must be AWB_LINKED or DISPATCHED

Body:
{
  "senderName": "Should Fail"
}

Assertions:
  - Status: 400
  - Body: { success: false, error: "Cannot update order: one or more parcels have already been AWB-linked or dispatched." }
