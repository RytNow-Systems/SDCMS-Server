### 2. List Upload Sessions
METHOD: GET
URL: {{baseUrl}}/bulk-uploads
HEADERS:
  Authorization: Bearer {{token}}
ASSERTIONS:
  res.status: eq 200
  res.body.success: eq true
  res.body.data[0].fileName: isString true
  res.body.data[0].totalRows: isNumber true

### 3. Get Specific Session Result
METHOD: GET
URL: {{baseUrl}}/bulk-uploads/:id
HEADERS:
  Authorization: Bearer {{token}}
ASSERTIONS:
  res.status: eq 200
  res.body.success: eq true
  res.body.data.session.fileName: isString true
  res.body.data.details: isArray true
